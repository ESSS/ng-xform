import { Component } from '@angular/core';

import { BaseDynamicFieldComponent } from '../field-components/base-dynamic-field.component';
import { TextField } from '../fields';


/**
 * Component to generate a bootstrap form field of general type
 *
 * :editing: Flag to control component state
 * :form: FormGroup containing the field
 * :field: Intance of field configurations
 */
@Component({
  selector: 'ng-xform-editable-label',
  templateUrl: './editable-label.component.html',
})
export class EditableLabelComponent extends BaseDynamicFieldComponent<TextField> {}

<div *ngIf="!isEditing && visible" id="{{field.key}}-label">
  <strong>{{ field.label }}</strong>
  <p class="text-muted">{{ form?.controls[field.key]?.value | valueSuffix }}</p>
</div>

<div *ngIf="isEditing && visible" id="{{elementId}}-div" class="form-group" [formGroup]="form" [ngClass]="displayFieldCss()">
  <label for="{{elementId}}">{{field.label}}</label> <ng-xform-optional-tag [show]="isOptional"></ng-xform-optional-tag>
  <input type="{{field.type}}" id="{{elementId}}" class="form-control" [formControlName]="elementId">
  <ng-xform-field-error-message [control]="control"></ng-xform-field-error-message>
</div>
