import { Component } from '@angular/core';

import { BaseDynamicFieldComponent } from '../field-components/base-dynamic-field.component';
import { AutocompleteField } from '../fields';


/**
 * Component to generate a bootstrap form field of general type
 *
 * :editing: Flag to control component state
 * :form: FormGroup containing the field
 * :field: Intance of field configurations
 */
@Component({
  selector: 'ng-xform-autocomplete-field',
  templateUrl: './autocomplete-field.component.html',
})
export class AutocompleteFieldComponent extends BaseDynamicFieldComponent<AutocompleteField> {}

<div *ngIf="!editing" id="{{field.key}}-label">
  <strong>{{ field.label }}</strong>
  <p class="text-muted">{{ form?.controls[field.key]?.value | valueSuffix }}</p>
</div>

<div *ngIf="editing" id="{{elementId}}-div" class="form-group" [formGroup]="form" [ngClass]="displayFieldCss()">
  <label for="{{elementId}}">{{field.label}}</label>
  <input id="{{elementId}}" class="form-control" [formControlName]="field.key"
    ngui-auto-complete
    [source]="field.source"
    [list-formatter]="field.listFormatter"
    [value-formatter]="field.valueFormatter"
    [accept-user-input]="false" />
  <ng-xform-field-error-message [control]="control"></ng-xform-field-error-message>
</div>
